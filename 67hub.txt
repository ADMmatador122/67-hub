local function notify(title, text)
    pcall(function()
        StarterGui:SetCore("SendNotification", {Title = title, Text = text, Duration = 3})
    end)
end

local function getRainbowColor()
    return Color3.fromHSV((tick() * rgbSpeed % 5) / 5, 1, 1)
end

local function tocarMusica(id)
    local RE = ReplicatedStorage:FindFirstChild("RE")
    if not RE then return end
    pcall(function() RE.PlayerToolEvent:FireServer("ToolMusicText", id) end)
    pcall(function() RE["1Player1sHous1e"]:FireServer("PickHouseMusicText", id) end)
    pcall(function() RE["1Player1sCa1r"]:FireServer("PickingCarMusicText", id) end)
    pcall(function() RE["1NoMoto1rVehicle1s"]:FireServer("PickingScooterMusicText", id) end)
end

local function fireServerRE(eventName, ...)
    local RE = ReplicatedStorage:FindFirstChild("RE")
    if RE and RE:FindFirstChild(eventName) then
        pcall(function() RE[eventName]:FireServer(...) end)
    end
end

local function findPlayerByPartialName(partial)
    partial = partial:lower()
    for _, p in pairs(Players:GetPlayers()) do
        if p ~= lp and p.Name:lower():sub(1, #partial) == partial then return p end
    end
    return nil
end

local function createSlider(text, minV, maxV, defV, callback)
    local f = createFrame(58)
    local lbl = Instance.new("TextLabel")
    lbl.Size = UDim2.new(1, -10, 0, 18)
    lbl.Position = UDim2.new(0, 10, 0, 4)
    lbl.BackgroundTransparency = 1
    lbl.Text = text .. ": " .. defV
    lbl.TextColor3 = Color3.new(0.85, 0.85, 0.85)
    lbl.Font = Enum.Font.Gotham
    lbl.TextSize = 12
    lbl.TextXAlignment = Enum.TextXAlignment.Left
    lbl.Parent = f
    local bg = Instance.new("Frame")
    bg.Size = UDim2.new(1, -20, 0, 14)
    bg.Position = UDim2.new(0, 10, 0, 28)
    bg.BackgroundColor3 = Color3.fromRGB(50, 50, 62)
    bg.BorderSizePixel = 0
    bg.Parent = f
    Instance.new("UICorner", bg).CornerRadius = UDim.new(1, 0)
    local fill = Instance.new("Frame")
    fill.Size = UDim2.new((defV - minV) / (maxV - minV), 0, 1, 0)
    fill.BackgroundColor3 = Color3.fromRGB(220, 50, 50)
    fill.BorderSizePixel = 0
    fill.Parent = bg
    Instance.new("UICorner", fill).CornerRadius = UDim.new(1, 0)
    local dragging = false
    local function update(pos)
        local rel = math.clamp((pos.X - bg.AbsolutePosition.X) / bg.AbsoluteSize.X, 0, 1)
        local val = math.floor(minV + rel * (maxV - minV))
        fill.Size = UDim2.new(rel, 0, 1, 0)
        lbl.Text = text .. ": " .. val
        callback(val)
    end
    bg.InputBegan:Connect(function(i)
        if i.UserInputType == Enum.UserInputType.MouseButton1 or i.UserInputType == Enum.UserInputType.Touch then
            dragging = true update(i.Position)
        end
    end)
    UserInputService.InputEnded:Connect(function() dragging = false end)
    UserInputService.InputChanged:Connect(function(i)
        if dragging and (i.UserInputType == Enum.UserInputType.MouseMovement or i.UserInputType == Enum.UserInputType.Touch) then
            update(i.Position)
        end
    end)
    return f
end

local function getESPColor()
    if espColor == "RGB" then return getRainbowColor()
    elseif espColor == "Vermelho" then return Color3.new(1,0,0)
    elseif espColor == "Verde" then return Color3.new(0,1,0)
    elseif espColor == "Azul" then return Color3.new(0,0.5,1)
    elseif espColor == "Amarelo" then return Color3.new(1,1,0)
    elseif espColor == "Rosa" then return Color3.new(1,0.5,0.8)
    elseif espColor == "Roxo" then return Color3.new(0.5,0,0.5)
    elseif espColor == "Branco" then return Color3.new(1,1,1)
    elseif espColor == "Preto" then return Color3.new(0,0,0)
    end
    return Color3.new(1,1,1)
end

local function removeESP(p)
    if billboardGuis[p] then billboardGuis[p]:Destroy() billboardGuis[p] = nil end
end
local function updateESP(p)
    if p == lp or not espEnabled then return end
    local char = p.Character
    if char and char:FindFirstChild("Head") then
        if billboardGuis[p] then billboardGuis[p]:Destroy() end
        local bill = Instance.new("BillboardGui")
        bill.Size = UDim2.new(0, 200, 0, 40)
        bill.StudsOffset = Vector3.new(0, 3, 0)
        bill.AlwaysOnTop = true
        bill.Parent = char.Head
        local tl = Instance.new("TextLabel")
        tl.Size = UDim2.new(1,0,1,0)
        tl.BackgroundTransparency = 1
        tl.Text = p.Name .. " | " .. p.AccountAge .. "d"
        tl.TextColor3 = getESPColor()
        tl.Font = Enum.Font.SourceSansBold
        tl.TextSize = 14
        tl.Parent = bill
        billboardGuis[p] = bill
    end
end

local espConn = nil
createToggle("ESP Ativado", false, function(state)
    espEnabled = state
    if state then
        for _, p in pairs(Players:GetPlayers()) do updateESP(p) end
        espConn = RunService.Heartbeat:Connect(function()
            for _, p in pairs(Players:GetPlayers()) do
                if espEnabled then
                    if not billboardGuis[p] then updateESP(p)
                    elseif billboardGuis[p] and billboardGuis[p]:FindFirstChild("TextLabel") then
                        billboardGuis[p].TextLabel.TextColor3 = getESPColor()
                    end
                end
            end
        end)
    else
        if espConn then espConn:Disconnect() espConn = nil end
        for _, p in pairs(Players:GetPlayers()) do removeESP(p) end
