local function tocarMusica(id)
    local RE = ReplicatedStorage:FindFirstChild("RE")
    if not RE then return end
    pcall(RE.PlayerToolEvent.FireServer, RE.PlayerToolEvent, "ToolMusicText", id)
    pcall(RE["1Player1sHous1e"].FireServer, RE["1Player1sHous1e"], "PickHouseMusicText", id)
    pcall(RE["1Player1sCa1r"].FireServer, RE["1Player1sCa1r"], "PickingCarMusicText", id)
    pcall(RE["1NoMoto1rVehicle1s"].FireServer, RE["1NoMoto1rVehicle1s"], "PickingScooterMusicText", id)
end

local function getRainbowColor()
    local h = (tick() * rgbSpeed % 5) / 5
    return Color3.fromHSV(h, 1, 1)
end

local function fireServerRGB(eventName, ...)
    local RE = ReplicatedStorage:FindFirstChild("RE")
    if RE and RE:FindFirstChild(eventName) then
        pcall(RE[eventName].FireServer, RE[eventName], ...)
    end
end

local function findPlayerByPartialName(partial)
    partial = partial:lower()
    for _, player in pairs(Players:GetPlayers()) do
        if player ~= LocalPlayer and player.Name:lower():sub(1, #partial) == partial then
            return player
        end
    end
    return nil
end

local function headsitOnPlayer(targetPlayer)
    local character = LocalPlayer.Character or LocalPlayer.CharacterAdded:Wait()
    local humanoid = character:FindFirstChildOfClass("Humanoid")
    if not targetPlayer.Character or not targetPlayer.Character:FindFirstChild("Head") then return false end
    local targetHead = targetPlayer.Character.Head
    local localRoot = character:FindFirstChild("HumanoidRootPart")
    if not localRoot then return false end
    localRoot.CFrame = targetHead.CFrame * CFrame.new(0, 2.2, 0)
    for _, v in pairs(localRoot:GetChildren()) do if v:IsA("WeldConstraint") then v:Destroy() end end
    local weld = Instance.new("WeldConstraint")
    weld.Part0 = localRoot
    weld.Part1 = targetHead
    weld.Parent = localRoot
    if humanoid then humanoid.Sit = true end
    return true
end

local function removeHeadsit()
    local character = LocalPlayer.Character or LocalPlayer.CharacterAdded:Wait()
    local humanoid = character:FindFirstChildOfClass("Humanoid")
    local localRoot = character:FindFirstChild("HumanoidRootPart")
    if localRoot then
        for _, v in pairs(localRoot:GetChildren()) do if v:IsA("WeldConstraint") then v:Destroy() end end
    end
    if humanoid then humanoid.Sit = false end
end

-- ESP
local function getESPColor()
    if espColor == "RGB" then return getRainbowColor()
    elseif espColor == "Preto" then return Color3.new(0,0,0)
    elseif espColor == "Branco" then return Color3.new(1,1,1)
    elseif espColor == "Vermelho" then return Color3.new(1,0,0)
    elseif espColor == "Verde" then return Color3.new(0,1,0)
    elseif espColor == "Azul" then return Color3.new(0,0.5,1)
    elseif espColor == "Amarelo" then return Color3.new(1,1,0)
    elseif espColor == "Rosa" then return Color3.new(1,0.5,0.8)
    elseif espColor == "Roxo" then return Color3.new(0.5,0,0.5)
    end
    return Color3.new(1,1,1)
end

local function updateESP(player)
    if player == LocalPlayer or not espEnabled then return end
    local character = player.Character
    if character and character:FindFirstChild("Head") then
        local head = character.Head
        if billboardGuis[player] then billboardGuis[player]:Destroy() end
        local bill = Instance.new("BillboardGui")
        bill.Size = UDim2.new(0, 200, 0, 40)
        bill.StudsOffset = Vector3.new(0, 2.5, 0)
        bill.AlwaysOnTop = true
        bill.Parent = head
        local text = Instance.new("TextLabel")
        text.Size = UDim2.new(1,0,1,0)
        text.BackgroundTransparency = 1
        text.Text = player.Name .. " | " .. player.AccountAge .. "d"
        text.TextColor3 = getESPColor()
        text.Font = Enum.Font.SourceSansBold
        text.TextSize = 14
        text.Parent = bill
        billboardGuis[player] = bill
    end
end

local function removeESP(player)
    if billboardGuis[player] then billboardGuis[player]:Destroy() end
    billboardGuis[player] = nil
end

local function createSlider(parent, text, minVal, maxVal, defaultVal, callback)
    local frame = Instance.new("Frame")
    frame.Size = UDim2.new(1, -8, 0, 55)
    frame.BackgroundTransparency = 1
    frame.Parent = parent
    local lbl = Instance.new("TextLabel")
    lbl.Size = UDim2.new(1,0,0,20)
    lbl.Text = text .. ": " .. defaultVal
    lbl.TextColor3 = Color3.new(1,1,1)
    lbl.BackgroundTransparency = 1
    lbl.Font = Enum.Font.SourceSans
    lbl.TextSize = 13
    lbl.Parent = frame
    local sliderBg = Instance.new("Frame")
    sliderBg.Size = UDim2.new(1, 0, 0, 20)
    sliderBg.Position = UDim2.new(0, 0, 0, 24)
    sliderBg.BackgroundColor3 = Color3.fromRGB(60,60,70)
    sliderBg.BorderSizePixel = 0
    local sliderCorner = Instance.new("UICorner")
    sliderCorner.CornerRadius = UDim.new(1,0)
    sliderCorner.Parent = sliderBg
    sliderBg.Parent = frame
    local fill = Instance.new("Frame")
    fill.Size = UDim2.new((defaultVal-minVal)/(maxVal-minVal), 0, 1, 0)
    fill.BackgroundColor3 = Color3.fromRGB(220,50,50)
    fill.BorderSizePixel = 0
    fill.Parent = sliderBg
    local dragging = false
    local function updateSlider(inputPos)
        local rel = (inputPos.X - sliderBg.AbsolutePosition.X) / sliderBg.AbsoluteSize.X
        rel = math.clamp(rel, 0, 1)
        local val = minVal + rel * (maxVal - minVal)
        fill.Size = UDim2.new(rel, 0, 1, 0)
        lbl.Text = text .. ": " .. math.floor(val)
        callback(math.floor(val))
    end
    sliderBg.InputBegan:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
            dragging = true
            updateSlider(input.Position)
        end
    end)
    UserInputService.InputEnded:Connect(function(input)
        dragging = false
    end)
    UserInputService.InputChanged:Connect(function(input)
        if dragging and (input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch) then
            updateSlider(input.Position)
        end
    end)
    return frame
end

local function createDropdown(parent, title, options, callback)
    local frame = Instance.new("Frame")
    frame.Size = UDim2.new(1, -8, 0, 50)
    frame.BackgroundTransparency = 1
    frame.Parent = parent
    local lbl = Instance.new("TextLabel")
    lbl.Size = UDim2.new(1,0,0,20)
    lbl.Text = title
    lbl.TextColor3 = Color3.new(1,1,1)
    lbl.BackgroundTransparency = 1
    lbl.Font = Enum.Font.SourceSans
    lbl.TextSize = 13
    lbl.Parent = frame
    local btn = Instance.new("TextButton")
    btn.Size = UDim2.new(0, 160, 0, 28)
    btn.Position = UDim2.new(0, 0, 0, 22)
    btn.BackgroundColor3 = Color3.fromRGB(55,55,65)
    btn.Text = options[1] or "Selecione"
    btn.TextColor3 = Color3.new(1,1,1)
    btn.BorderSizePixel = 0
    local corner = Instance.new("UICorner")
    corner.CornerRadius = UDim.new(0, 6)
    corner.Parent = btn
    btn.Parent = frame
    local dropdown = Instance.new("Frame")
    dropdown.Size = UDim2.new(0, 160, 0, 28 * #options)
    dropdown.Position = UDim2.new(0, 0, 0, 50)
    dropdown.BackgroundColor3 = Color3.fromRGB(45,45,55)
    dropdown.Visible = false
    dropdown.ZIndex = 10
    dropdown.Parent = frame
    local listLayout = Instance.new("UIListLayout")
    listLayout.Padding = UDim.new(0, 1)
    listLayout.Parent = dropdown
    for _, opt in ipairs(options) do
        local optBtn = Instance.new("TextButton")
        optBtn.Size = UDim2.new(1,0,0,28)
        optBtn.BackgroundColor3 = Color3.fromRGB(55,55,65)
        optBtn.Text = opt
        optBtn.TextColor3 = Color3.new(1,1,1)
        optBtn.BorderSizePixel = 0
        optBtn.Parent = dropdown
        optBtn.MouseButton1Click:Connect(function()
            btn.Text = opt
            dropdown.Visible = false
            callback(opt)
        end)
    end
    btn.MouseButton1Click:Connect(function()
        dropdown.Visible = not dropdown.Visible
    end)
    return frame
end

local carCat = createCategory("🚗 CARROS")

createButton(carCat, "📦 Trazer todos os carros", function()
    local pos = LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("HumanoidRootPart") and LocalPlayer.Character.HumanoidRootPart.Position
    if pos then
        for _, veh in pairs(workspace.Vehicles:GetChildren()) do
            if veh:IsA("Model") and veh.PrimaryPart then
                veh:SetPrimaryPartCFrame(CFrame.new(pos + Vector3.new(math.random(-10,10), 2, math.random(-10,10))))
            end
        end
    end
end)
createInput(carCat, "Velocidade do seu carro (ex: 80)", function(value)
    local speed = tonumber(value) or 50
    local car = workspace.Vehicles:FindFirstChild(LocalPlayer.Name.."Car")
    if car and car.Body and car.Body:FindFirstChild("VehicleSeat") then
        car.Body.VehicleSeat.TopSpeed.Value = speed
        StarterGui:SetCore("SendNotification", {Title = "Velocidade", Text = tostring(speed), Duration = 2})
    end
end)
local carSelectCat = createCategory("🚙 Selecionar Carro")
local selectedCarName = nil
local function getCarList()
    local cars = {}
    local veh = workspace:FindFirstChild("Vehicles")
    if veh then
        for _, c in pairs(veh:GetChildren()) do
            if c.Name:match("Car$") then table.insert(cars, c.Name) end
        end
    end
    if #cars == 0 then cars = {"Nenhum carro"} end
    return cars
end
local carDropdown = createDropdown(carSelectCat, "Selecione", getCarList(), function(v) selectedCarName = v end)
createButton(carSelectCat, "🔄 Atualizar lista", function()
    carDropdown.Parent:Destroy()
    carDropdown = createDropdown(carSelectCat, "Selecione", getCarList(), function(v) selectedCarName = v end)
end)
createButton(carSelectCat, "📥 Trazer carro selecionado", function()
    if selectedCarName and selectedCarName ~= "Nenhum carro" then
        local veh = workspace.Vehicles:FindFirstChild(selectedCarName)
        if veh and veh.PrimaryPart then
            local pos = LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("HumanoidRootPart") and LocalPlayer.Character.HumanoidRootPart.Position
            if pos then veh:SetPrimaryPartCFrame(CFrame.new(pos + Vector3.new(5,0,5))) end
        end
    end
end)
